1"""Core methylation extraction functions for bam2tensor. 

2 

3This module contains the main function for extracting methylation data from 

4BAM files: extract_methylation_data_from_bam(). This function reads aligned 

5bisulfite-sequencing (BS-seq) and enzymatic methylation sequencing (EM-seq) 

6reads and generates a sparse matrix representation of methylation states at 

7each CpG site. 

8 

9The extraction process: 

10 1. Iterates through each chromosome in the genome embedding 

11 2. For each aligned read that passes quality filters: 

12 a. Identifies which CpG sites fall within the read's alignment 

13 b. Determines the bisulfite strand (forward or reverse) 

14 c. Extracts methylation state at each covered CpG 

15 3. Builds a sparse COO matrix with the results 

16 

17Methylation State Encoding: 

18 - 1: Methylated (cytosine remained as C after bisulfite treatment) 

19 - 0: Unmethylated (cytosine converted to T by bisulfite treatment) 

20 - -1: No data (indel, SNV, or other non-C/T base at CpG position) 

21 

22Supported Aligners: 

23 The module supports BAM files from aligners that provide strand information: 

24 - Bismark: Uses the XM tag (Z=methylated CpG, z=unmethylated CpG) 

25 - Biscuit: Uses the YD tag (f=forward, r=reverse) 

26 - bwameth: Uses the YD tag (f=forward, r=reverse), identical to Biscuit 

27 - gem3/Blueprint: Uses the XB tag (C=forward, G=reverse) 

28 

29Example: 

30 >>> # xdoctest: +SKIP 

31 >>> from bam2tensor.embedding import GenomeMethylationEmbedding 

32 >>> from bam2tensor.functions import extract_methylation_data_from_bam 

33 >>> 

34 >>> # Create genome embedding 

35 >>> embedding = GenomeMethylationEmbedding( 

36 ... genome_name="hg38", 

37 ... expected_chromosomes=["chr1", "chr2"], 

38 ... fasta_source="reference.fa", 

39 ... ) 

40 >>> 

41 >>> # Extract methylation data 

42 >>> result = extract_methylation_data_from_bam( 

43 ... input_bam="sample.bam", 

44 ... genome_methylation_embedding=embedding, 

45 ... quality_limit=20, 

46 ... ) 

47 >>> 

48 >>> print(f"Extracted {result.matrix.shape[0]} reads, {result.matrix.nnz} data points") 

49"""

83def count_non_cpg_retained_xm(xm_tag: str) -> int: 

84 """Count retained non-CpG cytosines in a Bismark XM methylation string. 

85 

86 Bismark's ``XM`` tag encodes per-base methylation context. Uppercase 

87 letters indicate a cytosine that remained as ``C`` in the read 

88 (i.e., was *not* converted by bisulfite/EM-seq treatment). ``H``, 

89 ``X`` and ``U`` correspond to retained cytosines in CHH, CHG and 

90 unknown-context positions respectively. A high count of these on a 

91 single read is a strong signal of incomplete conversion. 

92 

93 Args: 

94 xm_tag: The value of a read's Bismark ``XM`` tag. 

95 

96 Returns: 

97 The count of ``H``, ``X`` and ``U`` characters in ``xm_tag``. 

98 

99 Example: 

100 >>> count_non_cpg_retained_xm("..Z..hhh..HHH..z..") 

101 3 

102 """ 

103 return xm_tag.count("H") + xm_tag.count("X") + xm_tag.count("U")

106def count_non_cpg_retained_reference( 

107 aligned_segment: pysam.AlignedSegment, 

108 is_reverse_parent_strand: bool, 

109) -> int: 

110 """Count retained non-CpG bases validated against the reference. 

111 

112 For a correctly bisulfite- or EM-seq-converted read, every 

113 non-CpG cytosine on the parent strand should have been converted. 

114 On the forward-parent strand that means every non-CpG ``C`` in the 

115 reference should appear as ``T`` in the read; on the reverse-parent 

116 strand every non-CpG ``G`` should appear as ``A``. Positions where 

117 the read still carries the unconverted base *and* the reference 

118 genuinely has a ``C``/``G`` (i.e., the mismatch is not a SNP) count 

119 as retained. 

120 

121 This is a faithful port of the logic in 

122 ``nebiolabs/mark-nonconverted-reads``, re-using the read's existing 

123 ``MD`` tag (via :py:meth:`pysam.AlignedSegment.get_aligned_pairs` 

124 with ``with_seq=True``) instead of requiring a separate reference 

125 FASTA. 

126 

127 Args: 

128 aligned_segment: A pysam aligned read. Must carry an ``MD`` 

129 tag; BAMs produced by Bismark, Biscuit, bwameth and gem3 

130 all set this tag by default. 

131 is_reverse_parent_strand: ``True`` if the read derives from the 

132 reverse (OB/CTOB) bisulfite parent strand, ``False`` for 

133 the forward (OT/CTOT) strand. 

134 

135 Returns: 

136 The number of reference-validated retained non-CpG 

137 cytosines (or guanines, for the reverse parent strand). Returns 

138 ``0`` when the read has no sequence or no ``MD`` tag is present. 

139 """ 

140 if aligned_segment.query_sequence is None: 140 ↛ 141line 140 didn't jump to line 141 because the condition on line 140 was never true

141 return 0 

142 

143 try: 

144 pairs = aligned_segment.get_aligned_pairs(matches_only=True, with_seq=True) 

145 except ValueError: 

146 # MD tag missing — cannot validate against reference. 

147 return 0 

148 

149 # Map ref_pos → reference base (uppercase) for CpG-context lookup. 

150 # matches_only=True guarantees query_pos, ref_pos, ref_base are all set. 

151 ref_pos_to_base = {rpos: rb.upper() for _, rpos, rb in pairs} 

152 

153 # On match, pysam returns ref_base uppercase (query matches ref). 

154 # On mismatch (SNP), it returns lowercase. We only care about matches 

155 # where ref is C/G — those are genuine retained, non-converted bases. 

156 target = "G" if is_reverse_parent_strand else "C" 

157 

158 count = 0 

159 for _, rpos, ref_base in pairs: 

160 if ref_base != target: 

161 # Not a match, or ref is not C/G. This rejects SNPs (lowercase) 

162 # and converted positions (read has T/A, match has different base). 

163 continue 

164 # Exclude CpG context: on forward strand, next ref base == G; 

165 # on reverse strand, previous ref base == C. 

166 if is_reverse_parent_strand: 166 ↛ 167line 166 didn't jump to line 167 because the condition on line 166 was never true

167 if ref_pos_to_base.get(rpos - 1) == "C": 

168 continue 

169 else: 

170 if ref_pos_to_base.get(rpos + 1) == "G": 

171 continue 

172 count += 1 

173 

174 return count

177def is_em_overconversion_read( 

178 read_cpg_states: list[int], 

179 min_cpgs: int, 

180) -> bool: 

181 """Identify reads flagged as EM-seq fragment-level over-conversion. 

182 

183 Loyfer et al. (bioRxiv 2026.03.24.713040) report that EM-seq 

184 produces a reproducible ~1–2.5% of multi-CpG fragments that appear 

185 fully unmethylated across every covered CpG, driven by failed TET 

186 protection and subsequent APOBEC hyper-conversion of an entire 

187 molecule. At constitutively methylated loci these reads are purely 

188 technical. Without a per-region methylation prior, the simplest 

189 correction consistent with their observation is: drop reads whose 

190 covered CpGs are all called unmethylated *and* cover at least 

191 ``min_cpgs`` sites (the paper's Fig. 1C regime where the artifact 

192 diverges clearly from WGBS). 

193 

194 This heuristic also drops genuinely fully-unmethylated biological 

195 fragments, so callers should opt in only when the downstream 

196 application can tolerate that trade-off. 

197 

198 Args: 

199 read_cpg_states: Per-CpG methylation state values for a single 

200 read, in column-order, using the bam2tensor encoding 

201 (``1``=methylated, ``0``=unmethylated, ``-1``=no data). 

202 min_cpgs: Minimum number of covered CpGs required to apply the 

203 filter. Reads with fewer covered CpGs are never flagged. 

204 

205 Returns: 

206 ``True`` when the read has at least ``min_cpgs`` covered CpGs 

207 and every covered CpG is called unmethylated (value ``0``). 

208 ``-1`` (no-data) values do not count as unmethylated. 

209 

210 Example: 

211 >>> is_em_overconversion_read([0, 0, 0], min_cpgs=3) 

212 True 

213 >>> is_em_overconversion_read([0, 0, 1], min_cpgs=3) 

214 False 

215 >>> is_em_overconversion_read([0, 0], min_cpgs=3) 

216 False 

217 """ 

218 if len(read_cpg_states) < min_cpgs: 

219 return False 

220 return all(state == 0 for state in read_cpg_states)

223def detect_aligner(input_bam: str, sample_size: int = 1000) -> str: 

224 """Detect the aligner used to produce a BAM file by checking read tags. 

225 

226 Samples up to ``sample_size`` reads from the BAM file and checks for 

227 aligner-specific tags: ``XM`` (Bismark), ``YD`` (Biscuit/bwameth), 

228 or ``XB`` (gem3/Blueprint). 

229 

230 Args: 

231 input_bam: Path to an indexed BAM file. 

232 sample_size: Maximum number of reads to examine. Only primary, 

233 non-duplicate, non-QC-failed reads are considered. 

234 

235 Returns: 

236 A human-readable string identifying the detected aligner, such as 

237 ``"Bismark (XM tag)"`` or ``"Unknown"``. 

238 """ 

239 try: 

240 bam = pysam.AlignmentFile(input_bam, "rb", require_index=True) # type: ignore 

241 except (FileNotFoundError, ValueError): 

242 return "Unknown (could not open BAM)" 

243 

244 checked = 0 

245 for read in bam.fetch(): 

246 if read.flag & _SKIP_FLAGS: 

247 continue 

248 if read.has_tag("XM"): 

249 bam.close() 

250 return "Bismark (XM tag)" 

251 if read.has_tag("YD"): 

252 bam.close() 

253 return "Biscuit/bwameth (YD tag)" 

254 if read.has_tag("XB"): 

255 bam.close() 

256 return "gem3/Blueprint (XB tag)" 

257 checked += 1 

258 if checked >= sample_size: 

259 break 

260 

261 bam.close() 

262 return "Unknown (no XM/YD/XB tags found)"

265def check_chromosome_overlap( 

266 bam_references: tuple[str, ...] | list[str], 

267 embedding_chromosomes: list[str], 

268) -> None: 

269 """Check that BAM and embedding chromosome names overlap. 

270 

271 Detects the common mistake where a BAM file uses one chromosome naming 

272 convention (e.g., ``1``, ``2``, ``3`` from Ensembl) while the reference 

273 embedding uses another (e.g., ``chr1``, ``chr2``, ``chr3`` from UCSC). 

274 If no overlap is found but adding or stripping a ``chr`` prefix would 

275 create overlap, raises a clear error with guidance. 

276 

277 Args: 

278 bam_references: Tuple or list of chromosome/contig names from the 

279 BAM file header (e.g., from ``pysam.AlignmentFile.references``). 

280 embedding_chromosomes: List of chromosome names from the genome 

281 embedding (i.e., the expected chromosomes). 

282 

283 Raises: 

284 ValueError: If there is no overlap between BAM and embedding 

285 chromosome names, with a diagnostic message indicating 

286 whether a ``chr`` prefix mismatch is the likely cause. 

287 """ 

288 bam_set = set(bam_references) 

289 emb_set = set(embedding_chromosomes) 

290 

291 overlap = bam_set & emb_set 

292 if overlap: 

293 return 

294 

295 # Try adding "chr" prefix to BAM references 

296 bam_with_chr = {f"chr{name}" for name in bam_set} 

297 if bam_with_chr & emb_set: 

298 bam_example = next(iter(bam_set - emb_set)) 

299 emb_example = f"chr{bam_example}" 

300 raise ValueError( 

301 f"Chromosome name mismatch: BAM file uses '{bam_example}' " 

302 f"but the reference embedding expects '{emb_example}'. " 

303 f"Your BAM file appears to use Ensembl-style names (no 'chr' prefix) " 

304 f"while the embedding uses UCSC-style names ('chr' prefix). " 

305 f"Use --expected-chromosomes with matching names " 

306 f"(e.g., --expected-chromosomes '1,2,3,...')." 

307 ) 

308 

309 # Try stripping "chr" prefix from BAM references 

310 bam_stripped = {name[3:] for name in bam_set if name.startswith("chr")} 

311 if bam_stripped & emb_set: 

312 bam_example = next( 

313 name for name in bam_set if name.startswith("chr") and name[3:] in emb_set 

314 ) 

315 emb_example = bam_example[3:] 

316 raise ValueError( 

317 f"Chromosome name mismatch: BAM file uses '{bam_example}' " 

318 f"but the reference embedding expects '{emb_example}'. " 

319 f"Your BAM file uses UCSC-style names ('chr' prefix) " 

320 f"while the embedding uses Ensembl-style names (no 'chr' prefix). " 

321 f"Use --expected-chromosomes with matching names " 

322 f"(e.g., --expected-chromosomes 'chr1,chr2,chr3,...')." 

323 ) 

324 

325 # No overlap even with prefix changes -- completely different names 

326 bam_sample = sorted(bam_set)[:3] 

327 emb_sample = sorted(emb_set)[:3] 

328 raise ValueError( 

329 f"No overlapping chromosome names between BAM file and reference " 

330 f"embedding. BAM contains: {bam_sample}... " 

331 f"Embedding expects: {emb_sample}... " 

332 f"Ensure the reference FASTA matches the genome used for alignment " 

333 f"and that --expected-chromosomes is set correctly." 

334 )

337def extract_methylation_data_from_bam( 

338 input_bam: str, 

339 genome_methylation_embedding: GenomeMethylationEmbedding, 

340 quality_limit: int = 20, 

341 filter_non_converted: bool = False, 

342 non_converted_threshold: int = 3, 

343 filter_em_overconversion: bool = False, 

344 em_overconversion_min_cpgs: int = 3, 

345 verbose: bool = False, 

346 debug: bool = False, 

347) -> ExtractionResult: 

348 """Extract read-level CpG methylation data from a BAM file. 

349 

350 Parses a bisulfite-sequencing or EM-seq BAM file and extracts methylation 

351 states at each CpG site covered by aligned reads. Returns a sparse matrix 

352 where each row represents a unique read and each column represents a CpG 

353 site. 

354 

355 The function applies several filters to reads: 

356 - Mapping quality must be >= quality_limit (default 20) 

357 - Duplicate reads are excluded 

358 - QC-failed reads are excluded 

359 - Secondary and supplementary alignments are excluded 

360 - For Biscuit/bwameth/gem3: only parent-strand reads are processed 

361 - For Bismark: all reads are processed (XM tag has pre-resolved calls) 

362 

363 Two additional, opt-in per-read filters are available: 

364 - Non-converted reads (``filter_non_converted``): drops reads with 

365 too many retained non-CpG cytosines, the hallmark of incomplete 

366 bisulfite/EM-seq conversion. Ports the logic of 

367 ``nebiolabs/mark-nonconverted-reads``. 

368 - EM-seq fragment-level over-conversion 

369 (``filter_em_overconversion``): drops reads whose covered CpGs 

370 are all called unmethylated, a heuristic for the EM-seq 

371 artifact described by Loyfer et al. 

372 (bioRxiv 2026.03.24.713040). 

373 

374 Two extraction paths are supported, detected automatically per-read: 

375 

376 **Bismark path** (XM tag present): 

377 1. Uses binary search to find CpG sites within the read 

378 2. Reads the XM methylation call string directly 

379 3. Records: Z=methylated(1), z=unmethylated(0), other CpG=-1 

380 

381 **Biscuit/bwameth/gem3 path** (YD or XB tag): 

382 1. Checks the strand tag to filter daughter-strand reads 

383 2. Uses binary search to find CpG sites within the read 

384 3. Extracts the base at each CpG position from the read sequence 

385 4. Records: C=methylated(1), T=unmethylated(0), other=-1 

386 

387 Args: 

388 input_bam: Path to the input BAM file. The file must be indexed 

389 (have a corresponding .bam.bai file). 

390 genome_methylation_embedding: A GenomeMethylationEmbedding object 

391 that defines the CpG site positions and provides coordinate 

392 conversion methods. 

393 quality_limit: Minimum mapping quality (MAPQ) threshold for reads. 

394 Reads with MAPQ below this value are skipped. Default is 20, 

395 which excludes reads mapping to multiple locations equally well. 

396 filter_non_converted: If True, drop reads that carry at least 

397 ``non_converted_threshold`` retained non-CpG cytosines, a 

398 signature of incomplete bisulfite/EM-seq conversion. Default 

399 False. 

400 non_converted_threshold: Minimum count of retained non-CpG 

401 cytosines required for the non-converted filter to drop a 

402 read. Matches the NEB ``mark-nonconverted-reads`` default of 

403 3. 

404 filter_em_overconversion: If True, drop reads whose covered CpGs 

405 are all called unmethylated and cover at least 

406 ``em_overconversion_min_cpgs`` sites — the Loyfer et al. 

407 EM-seq fragment-level over-conversion heuristic. Default 

408 False. 

409 em_overconversion_min_cpgs: Minimum covered CpG count required 

410 before the over-conversion filter will drop a read. Matches 

411 the regime in Loyfer et al. Fig. 1C where the EM-seq 

412 artifact is clearly separable from WGBS. 

413 verbose: If True, display a progress bar and print the total read 

414 count. Useful for monitoring progress on large files. 

415 debug: If True, enable extensive validation and debug output. 

416 Prints per-read information and validates that each read is 

417 only processed once. Significantly slower. 

418 

419 Returns: 

420 An ExtractionResult named tuple with two fields: 

421 

422 - **matrix**: A scipy.sparse.coo_matrix with shape 

423 (n_reads, n_cpg_sites) where n_reads is the number of reads 

424 that passed filters and covered at least one CpG site, 

425 n_cpg_sites is genome_methylation_embedding.total_cpg_sites, 

426 and values are: 1 (methylated), 0 (unmethylated), -1 (no data). 

427 The matrix uses COO format for efficient construction; convert 

428 to CSR (tocsr()) for row slicing or CSC (tocsc()) for column 

429 slicing. 

430 - **tlen**: A 1-D numpy int32 array of shape (n_reads,) containing 

431 the signed template length (BAM TLEN field) for each read. 

432 Values are 0 for single-end reads or reads with unmapped mates. 

433 

434 Raises: 

435 FileNotFoundError: If the BAM file index (.bam.bai) is missing. 

436 The BAM file must be indexed with `samtools index`. 

437 

438 Example: 

439 >>> # xdoctest: +SKIP 

440 >>> from bam2tensor.embedding import GenomeMethylationEmbedding 

441 >>> from bam2tensor.functions import extract_methylation_data_from_bam 

442 >>> import scipy.sparse 

443 >>> 

444 >>> # Create embedding (or load from cache) 

445 >>> embedding = GenomeMethylationEmbedding( 

446 ... genome_name="hg38", 

447 ... expected_chromosomes=["chr1", "chr22"], 

448 ... fasta_source="GRCh38.fa", 

449 ... ) 

450 >>> 

451 >>> # Extract methylation data 

452 >>> result = extract_methylation_data_from_bam( 

453 ... input_bam="sample.bam", 

454 ... genome_methylation_embedding=embedding, 

455 ... quality_limit=30, # Stricter quality filter 

456 ... verbose=True, 

457 ... ) 

458 >>> 

459 >>> # Analyze results 

460 >>> print(f"Reads with CpG data: {result.matrix.shape[0]:,}") 

461 >>> print(f"Total CpG sites: {result.matrix.shape[1]:,}") 

462 >>> print(f"Data points: {result.matrix.nnz:,}") 

463 >>> 

464 >>> # Save to file 

465 >>> scipy.sparse.save_npz("sample.methylation.npz", result.matrix) 

466 

467 Note: 

468 The function processes chromosomes in the order they appear in 

469 genome_methylation_embedding.cpg_sites_dict. Reads are numbered 

470 sequentially as they are encountered across all chromosomes. 

471 

472 See Also: 

473 GenomeMethylationEmbedding: For creating the genome embedding. 

474 scipy.sparse.coo_matrix: Documentation on the output format. 

475 """ 

476 try: 

477 input_bam_object = pysam.AlignmentFile( # type: ignore # pylint: disable=no-member 

478 input_bam, "rb", require_index=True, threads=4 

479 ) 

480 except FileNotFoundError as exc: 

481 raise FileNotFoundError(f"Index missing for bam file?: {input_bam}") from exc 

482 

483 # Check for chromosome name mismatches (e.g. chr1 vs 1) 

484 check_chromosome_overlap( 

485 bam_references=input_bam_object.references, 

486 embedding_chromosomes=list(genome_methylation_embedding.cpg_sites_dict.keys()), 

487 ) 

488 

489 if verbose: 

490 print(f"\tTotal reads: {input_bam_object.mapped:,}\n") 

491 if debug: 

492 debug_read_name_to_row_number = {} 

493 

494 # Temporary storage for loading into a COO matrix 

495 # For a COO, we want three lists: 

496 # row: the read number (we'll store a read name -> ID dict perhaps?) 

497 # column: cpg # 

498 # data: methylation state (1 [methylated], 0 [unmethylated], and -1 [no data/snv/indel]) 

499 read_number = 0 

500 

501 coo_row = [] # Read number 

502 coo_col = [] # CpG number (embedding) 

503 coo_data = [] # Methylation state 

504 tlen_list: list[int] = [] # Template length (TLEN) per read 

505 

506 # This is slow, but we only run it once and store the results for later 

507 for chrom, cpg_sites in tqdm( 

508 genome_methylation_embedding.cpg_sites_dict.items(), 

509 disable=not verbose, 

510 ): 

511 # Ensure this chromosome exists in the BAM file 

512 try: 

513 iter_reads = input_bam_object.fetch(contig=chrom) 

514 except ValueError: 

515 # Chromosome not in BAM 

516 if verbose: 516 ↛ 518line 516 didn't jump to line 518 because the condition on line 516 was always true

517 tqdm.write(f"Chromosome {chrom} not found in BAM file, skipping.") 

518 continue 

519 

520 for aligned_segment in iter_reads: 

521 if aligned_segment.mapping_quality < quality_limit: 

522 continue 

523 

524 # Single bitwise check replaces 4 separate flag checks: 

525 # duplicate (0x400), qcfail (0x200), secondary (0x100), supplementary (0x800) 

526 if aligned_segment.flag & _SKIP_FLAGS: 

527 continue 

528 

529 # Per-read buffers. We only flush these into the global 

530 # coo_* arrays once the read passes all filters (including 

531 # the post-CpG EM over-conversion filter). 

532 read_cpg_cols: list[int] = [] 

533 read_cpg_data: list[int] = [] 

534 

535 # ============================================================ 

536 # Bismark path: XM tag contains pre-resolved methylation calls. 

537 # No strand filtering needed — Bismark already resolved strand 

538 # and encoded calls as Z (methylated CpG) / z (unmethylated CpG). 

539 # Both reads in a pair carry valid XM tags. 

540 # ============================================================ 

541 if aligned_segment.has_tag("XM"): 

542 xm_tag: str = aligned_segment.get_tag("XM") # type: ignore[assignment] 

543 

544 # Non-converted filter (Bismark): XM tag already encodes 

545 # retained non-CpG cytosines as H/X/U. Apply before any 

546 # CpG work so we bail as early as possible. 

547 if filter_non_converted: 

548 if count_non_cpg_retained_xm(xm_tag) >= non_converted_threshold: 

549 continue 

550 

551 # Find CpG sites covered by this read 

552 start_idx = bisect.bisect_left( 

553 cpg_sites, aligned_segment.reference_start + 1 

554 ) 

555 end_idx = bisect.bisect_right(cpg_sites, aligned_segment.reference_end) 

556 if start_idx >= end_idx: 

557 continue 

558 

559 cpgs_within_read_set = set(cpg_sites[start_idx:end_idx]) 

560 

561 # Map read positions to reference positions, filter to CpG sites
